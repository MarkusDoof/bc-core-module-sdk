45 static bc_tick_t _bc_module_relay_task(void *param)
46 {
47  bc_module_relay_t *self = param;
48 
49  switch(self->_state)
50  {
51  // Final state turning off the coil
52  case BC_MODULE_RELAY_TASK_STATE_DEMAGNETIZE:
53  // De-energize bistable relay coil - turn off
54  bc_tca9534a_write_port(&(self->_tca9534a), (1 << 6) | (1 << 4));
55  // Unregister task
56  bc_module_relay_scheduler_unregister(self);
57  return 0;
58  break;
59 
60  // Pulse mode - demagnetize the coil and set state to toggle the coil to other direction
61  case BC_MODULE_RELAY_TASK_STATE_DEMAGNETIZE_PULSE:
62  // De-energize bistable relay coil - turn off
63  bc_tca9534a_write_port(&(self->_tca9534a), (1 << 6) | (1 << 4));
64  // Set next task state
65  self->_state = BC_MODULE_RELAY_TASK_STATE_REVERSE_PULSE;
66  // Delay for set period of the pulse
67  return self->_pulse_length;
68  break;
69 
70  // Pulse mode - toggle the relay to another direction
71  case BC_MODULE_RELAY_TASK_STATE_REVERSE_PULSE:
72  bc_module_relay_toggle(self);
73  return 20;
74  break;
75 
76  default:
77  break;
78  }
79 
80  return 0;
