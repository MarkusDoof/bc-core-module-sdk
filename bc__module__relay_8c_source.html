16 static bool _bc_module_relay_hardware_init(bc_module_relay_t *self)
17 {
18  // Init i2C expander driver
19  if (!bc_tca9534a_init(&self->_tca9534a, BC_I2C_I2C0, BC_MODULE_RELAY_I2C_ADDRESS_DEFAULT))
20  {
21  return false;
22  }
23  // De-energize bistable relay coil - turn off
24  bc_tca9534a_write_port(&(self->_tca9534a), BC_MODULE_RELAY_POLARITY_NONE);
25  // Enable outputs
26  bc_tca9534a_set_port_direction(&self->_tca9534a, 0x00); // inverted: 0 = output
27  // Relay is bi-stable, in the begining we don't know the default state
28  self->_relay_state = BC_MODULE_RELAY_STATE_UNKNOWN;
29 
30  return true;
31 }

91 static bc_tick_t bc_module_relay_state_machine(bc_module_relay_t *self)
92 {
93  while(true)
94  {
95  switch (self->_state)
96  {
97 
98  case BC_MODULE_RELAY_TASK_STATE_IDLE:
99  // Handle Error
100  if(self->_error)
101  {
102  // Try to initialize relay module again
103  if (_bc_module_relay_hardware_init(self))
104  {
105  self->_error = false;
106  }
107  }
108 
109  // Handle commands
110  if (self->_command == BC_MODULE_RELAY_COMMAND_SET)
111  {
112  self->_command = BC_MODULE_RELAY_COMMAND_NONE;
113  self->_state = BC_MODULE_RELAY_TASK_STATE_SET;
114  continue;
115  }
116 
117  if (self->_command == BC_MODULE_RELAY_COMMAND_PULSE)
118  {
119  self->_command = BC_MODULE_RELAY_COMMAND_NONE;
120  self->_state = BC_MODULE_RELAY_TASK_STATE_PULSE;
121  continue;
122  }
123 
124  // Unregister task if no command is needed
125  bc_module_relay_scheduler_unregister(self);
126  return 0;
127 
128 
129  //
130  // Relay set start state
131  //
132  case BC_MODULE_RELAY_TASK_STATE_SET:
133  // Set relay to the selected polarity
134  _bc_module_relay_set_output(self, self->_desired_state);
135  self->_relay_state = self->_desired_state;
136 
137  self->_state = BC_MODULE_RELAY_TASK_STATE_SET_DEMAGNETIZE;
138  return 20;
139 
140  case BC_MODULE_RELAY_TASK_STATE_SET_DEMAGNETIZE:
141  // De-energize bistable relay coil - turn off
142  _bc_module_relay_set_output_disable(self);
143 
144  self->_state = BC_MODULE_RELAY_TASK_STATE_IDLE;
145  // Needs 100ms to let the capacitor on relay board to charge
146  return 100;
147 
148 
149  //
150  // Relay pulse start state
151  //
152  case BC_MODULE_RELAY_TASK_STATE_PULSE:
153  // Create pulse of the set polarity
154  _bc_module_relay_set_output(self, self->_desired_state);
155  self->_relay_state = self->_desired_state;
156 
157  self->_state = BC_MODULE_RELAY_TASK_STATE_PULSE_DEMAGNETIZE;
158  return 20;
159 
160  case BC_MODULE_RELAY_TASK_STATE_PULSE_DEMAGNETIZE:
161  // De-energize bistable relay coil - turn off
162  _bc_module_relay_set_output_disable(self);
163 
164  self->_state = BC_MODULE_RELAY_TASK_STATE_PULSE_REVERSE;
165  return self->_pulse_duration;
166 
167  case BC_MODULE_RELAY_TASK_STATE_PULSE_REVERSE:
168  // Change actual relay state to the oposite polarity
169  self->_relay_state = (self->_relay_state == BC_MODULE_RELAY_STATE_TRUE) ? BC_MODULE_RELAY_STATE_FALSE : BC_MODULE_RELAY_STATE_TRUE;
170  _bc_module_relay_set_output(self, self->_relay_state);
171 
172  self->_state = BC_MODULE_RELAY_TASK_STATE_PULSE_DEMAGNETIZE_2;
173  return 20;
174 
175  case BC_MODULE_RELAY_TASK_STATE_PULSE_DEMAGNETIZE_2:
176  // De-energize bistable relay coil - turn off
177  _bc_module_relay_set_output_disable(self);
178 
179  self->_state = BC_MODULE_RELAY_TASK_STATE_IDLE;
180  // Needs 100ms to let the capacitor on relay board to charge
181  return 100;
182 
183  default:
184  break;
185  }
186  }
187 }
